is_package_installed <- function(package_name) {
  return(package_name %in% installed.packages()[, "Package"])
}

# This function does several things. It is recommended to run it 
# at the beginning of every notebook.
# 1. install and load most commonly used packages
# 2. prints the current time and date
# 3. disables scientific notation
# 4. sets the html display to auto-margins at 100%
# 5. sets plot sizes to 8x8
notebook_setup <- function(packages_to_install = c("base64enc", "bigrquery", "caret", "corrplot", "devtools", "DHARMa",
                                                   "DMwR2", "doParallel", "dplyr", "emmeans", "fields", 
                                                   "foreach", "ggcorrplot", 
                                                   "ggplot2",  
                                                   "ggthemes", "googleCloudStorageR", "grDevices", 
                                                   "htmltools", "IRdisplay", "kableExtra", "knitr", 
                                                   "leaflet", "lmtest", "lqmm", "lubridate", "lwgeom", "MASS", 
                                                   "MatchIt", "mice", 
                                                   "ncdf4", "nngeo", "parallel", "patchwork", "performance", 
                                                   "plotly", "plotrix", 
                                                   "purrr", "questionr", "RColorBrewer", "RCurl", 
                                                   "RJSONIO", "raster", "readxl", "reshape2", "sandwich",
                                                   "scales", "sf", "shiny", "sjPlot", "stringr", "survey", 
                                                   "svMisc", "tableone", "tidycensus", "tidyr", 
                                                   "tidyverse", "tigris", "twang", "viridis", "webshot2", "yaml"), 
                           git_packages_to_install = c("kforthman/optLog", "kforthman/libRtheme"),
                           update_packages = FALSE,
                           plot.width=8, plot.height=8
){

  # Install CRAN packages if not already installed
  if(!is.null(packages_to_install)){
    
    for (package in packages_to_install) {
      if (!is_package_installed(package)) {
        install.packages(package)
      }
      if (update_packages){
          suppressWarnings(update.packages(oldPkgs = package, ask = FALSE))
      }
      suppressPackageStartupMessages(library(package, character.only = T))
    }
    
  }
  
  # Install git packages if not already installed
  if(!is.null(git_packages_to_install)){
    
    for (package in git_packages_to_install) {
      if (!is_package_installed(strsplit(package, "/")[[1]][2]) | update_packages) {
        install_github(package)
      }
      suppressPackageStartupMessages(library(strsplit(package, "/")[[1]][2], character.only = T))
    }
    
  }
  
  # Print the time
  Sys.setenv(TZ="America/Chicago")
  suppressWarnings({
    current_time <- Sys.time()
    formatted_time <- format(current_time, "%A, %B %d %Y\n%I:%M %p")
    cat(formatted_time)
  })
  
  # Disable scientific notation formatting
  options(scipen=999)
  
}

# Helper function to get the hash of a local file
get_local_file_hash <- function(filename) {
  file_hash <- digest::digest(object = filename, algo = "md5", file = TRUE)
  return(file_hash)
}

stat_smd <- function(var, treat, data){
    z <- data[[treat]]
    x <- data[[var]]
    xT <- x[z == 1]
    xC <- x[z == 0]
    
    # group means and sds
    m0 <- mean(xC)
    m1 <- mean(xT)
    s0 <- sd(xC)
    s1 <- sd(xT)

    # pooled SD
    s_pooled <- sqrt((s0^2 + s1^2)/2)

    # standardized difference
    smd_x <- (m1 - m0) / s_pooled
    smd_x
    
    return(smd_x)
}

stat_pval <- function(var, treat, data){ #  vname, "treatment", matched_data
    z <- data[[treat]]
    x <- data[[var]]
    xT <- x[z == 1]
    xC <- x[z == 0]
    # 2c. Decide which test to run
    if (is.numeric(x)) {
        # Two‐sample t‐test (Welch’s by default)
        test_obj <- try(t.test(xT, xC), silent = TRUE)
        # In very rare cases (e.g. all values identical) t.test might error; handle gracefully:
        if (inherits(test_obj, "htest")) {
            p_val <- test_obj$p.value
        } else {
            p_val <- NA_real_
        }
    } else {
        # Build a 2×2 table for chi‐square or Fisher’s exact
        tab <- table(x, z)
        # If any expected cell count < 5, use Fisher’s exact
        chi_obj <- try(chisq.test(tab, correct = FALSE), silent = TRUE)
        if (inherits(chi_obj, "htest") && all(chi_obj$expected >= 5)) {
            p_val <- chi_obj$p.value
        } else {
            fish_obj <- try(fisher.test(tab), silent = TRUE)
            p_val     <- if (inherits(fish_obj, "htest")) fish_obj$p.value else NA_real_
                  }
    }
    return(p_val)
}
